    <div id="membershipSection" style="display:none;">

      <label>Is your membership active? (yes/no): </label><br>

      <input type="text" id="activeMembership"><br><br>

        <label>Number of adults: <input type="number" id="memberAdults"></label><br>

      <label>Is group size up to 20? (yes/no): </label><br>

      <input type="text" id="groupSize"><br><br>


      <div id="lessThan20" style="display:none;">

        <p>Do the members qualify for:</p>

        <ul>

          <li>Income Assistance</li>

          <li>Oregon/Washington residency with ID</li>

        </ul>

        <label>Do they qualify? (yes/no): </label><br>

        <input type="text" id="qualified"><br>

        <label>Number of adults: <input type="number" id="adults"></label><br>

        <label>Number of children: <input type="number" id="children"></label><br><br>

      </div>


      <div id="moreThan20" style="display:none;">

        <label>Is your group non-profit? (yes/no): </label><br>

        <input type="text" id="nonProfit"><br>

        <label>Group size: <input type="number" id="groupCount"></label><br><br>

      </div>

    <button onclick="calculate()">Calculate Price</button>

    <h3 id="result"></h3>

  </div>


  <script>

    const adultFee = 26, childFee = 21;

    const flatRate = 13, sub = 4;

    const memberFees = { adult: 59, child: 39, guest: 79 };


    function calculate() {

      const membership = document.getElementById("membership").value.toLowerCase();

      let result = document.getElementById("result");


      if (membership === "yes") {

        const active = document.getElementById("activeMembership").value.toLowerCase();

        if (active === "yes") {

          result.innerText = "As an active member, your trip is FREE.";

          return;

        } else {

          const adults = parseInt(document.getElementById("memberAdults").value);

          const children = parseInt(document.getElementById("memberChildren").value);

          const guests = parseInt(document.getElementById("memberGuests").value);

          const price = (adults * memberFees.adult) + (children * memberFees.child) + (guests * memberFees.guest);

          result.innerText = `Your membership cost estimate is $${price}.`;

          return;

        }

      }


      const groupSize = document.getElementById("groupSize").value.toLowerCase();

      if (groupSize === "yes") {

        const qualified = document.getElementById("qualified").value.toLowerCase();

        const adults = parseInt(document.getElementById("adults").value);

        const children = parseInt(document.getElementById("children").value);

        const totalPeople = adults + children;

        let price = 0;


        if (qualified === "yes") {

          if (totalPeople <= sub) {

            price = totalPeople * flatRate;

          } else {

            let discountedAdults = Math.min(adults, sub);

            let discountedRemaining = sub - discountedAdults;

            let discountedChildren = Math.min(children, discountedRemaining);

            let remainingAdults = adults - discountedAdults;

            let remainingChildren = children - discountedChildren;

            price += (discountedAdults + discountedChildren) * flatRate;

            price += (remainingAdults * adultFee) + (remainingChildren * childFee);

          }

        } else {

          price = (adults * adultFee) + (children * childFee);

        }


        result.innerText = `Your group trip estimate is $${price}.`;

      } else {

        const nonProfit = document.getElementById("nonProfit").value.toLowerCase();

        const size = parseInt(document.getElementById("groupCount").value);

        const fixed = nonProfit === "yes" ? 13 : 19;

        const price = size * fixed;

        result.innerText = `Your group trip estimate is $${price}.`;

      }

    }


    // Show/hide sections based on inputs

    document.getElementById("membership").addEventListener("input", () => {

      let mem = document.getElementById("membership").value.toLowerCase();

      document.getElementById("membershipSection").style.display = (mem === "yes") ? "block" : "none";

      document.getElementById("groupSection").style.display = (mem === "no") ? "block" : "none";

    });


    document.getElementById("activeMembership").addEventListener("input", () => {

      let active = document.getElementById("activeMembership").value.toLowerCase();

      document.getElementById("inactiveMembershipInputs").style.display = (active === "no") ? "block" : "none";

    });


    document.getElementById("groupSize").addEventListener("input", () => {

      let val = document.getElementById("groupSize").value.toLowerCase();

      document.getElementById("lessThan20").style.display = (val === "yes") ? "block" : "none";

      document.getElementById("moreThan20").style.display = (val === "no") ? "block" : "none";

    });

  </script>
